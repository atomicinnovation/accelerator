import { Card, CardGrid, LinkCard } from '@astrojs/starlight/components'

<CardGrid stagger>
  <Card title="Filesystem phases" icon="document">
    Research, plan, and implement phases hand off through documents in
    `meta/`, not through a single ever-growing conversation — each phase
    starts with a fresh, focused context.
  </Card>
  <Card title="Bounded contexts" icon="rocket">
    Subagents do exploratory work in isolated contexts and return only
    summaries, keeping the main conversation lean and on-task.
  </Card>
  <Card title="Generated skill reference" icon="open-book">
    Every skill's reference page is generated from its `SKILL.md` source
    at build time, so the documentation can never drift from the plugin.
  </Card>
  <Card title="Visualiser" icon="laptop">
    A browser-based companion view renders your `meta/` documents — plans,
    research, work items — as a live, navigable site.
  </Card>
</CardGrid>
